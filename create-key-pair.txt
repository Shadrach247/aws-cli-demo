# create key pair
aws ec2 create-key-pair \
  --key-name EffectiveDevOpsAWS \
  --query "KeyMaterial" \
  --output text > ~/.ssh/EffectiveDevOpsAWS.pem


# example-2 - create key pair
aws ec2 create-key-pair \
    --key-name DevOps \
    --query "KeyMaterial" \
    --output text > ~/.ssh/DevOps.pem

# example-3 create key pair
aws ec2 create-key-pair \
  --key-name MyKeyPair \
  --query 'KeyMaterial' \
  --output text > MyKeyPair.pem

# describe key-pair
aws ec2 describe-key-pairs \
  --key-name DevOps

# cat key-pair
cat ~/.ssh/DevOps.pem

# set permission
chmod 400 ~/.ssh/DevOps.pem

#To delete a key pair
aws ec2 delete-key-pair --key-name MyKeyPair
